geo_properties <- list(
    scope = "usa",
    projection = list(type = 'albers usa'),
    showland = TRUE,
    showsubunits = TRUE,
    landcolor = toRGB("gray10"),
    showlakes = TRUE,
    lakecolor = toRGB("white")
)


map <- plot_geo(data) %>%
    add_markers(
        x = ~longitude,
        y = ~latitude,
        color = ~avg_rating,
        colors ='RdBu',
        text = ~hover,
        hoverinfo = "text",
        marker = list(
          symbol = "circle",
          size = 8,
          line = list(color = "black", width = 1),
          opacity = 0.8
        )
      ) %>%
    layout(geo = geo_properties)
